自定义工具类 (Custom Utilities)

自定义插件

// tailwind.config.js
const plugin = require('tailwindcss/plugin')

module.exports = {
  plugins: [
    plugin(function({ addUtilities }) {
      addUtilities({
        '.scrollbar-hide': {
          '-ms-overflow-style': 'none',
          'scrollbar-width': 'none',
          '&::-webkit-scrollbar': {
            display: 'none'
          }
        },
        '.scrollbar-default': {
          '-ms-overflow-style': 'auto',
          'scrollbar-width': 'auto',
          '&::-webkit-scrollbar': {
            display: 'block'
          }
        }
      })
    })
  ]
}

<div class="h-32 overflow-y-scroll scrollbar-hide">
  <div class="h-64">这个容器的滚动条被隐藏了</div>
</div>

高级动画 (Advanced Animations)

自定义动画

// tailwind.config.js
module.exports = {
  theme: {
    extend: {
      animation: {
        'spin-slow': 'spin 3s linear infinite',
        'bounce-slow': 'bounce 2s infinite',
        'pulse-slow': 'pulse 3s cubic-bezier(0.4, 0, 0.6, 1) infinite',
      }
    }
  }
}

<div class="flex space-x-4">
  <div class="w-16 h-16 bg-blue-500 rounded animate-spin-slow"></div>
  <div class="w-16 h-16 bg-green-500 rounded animate-bounce-slow"></div>
  <div class="w-16 h-16 bg-red-500 rounded animate-pulse-slow"></div>
</div>
